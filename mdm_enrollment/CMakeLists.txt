cmake_minimum_required(VERSION 3.10)
project("mdm_extension")

function(mdmExtensionMain)
    addOsqueryExtensionEx("MDMStatusTablePlugin" "table" "mdm_enrollment_status"
        SOURCES mdm_status_extension.cpp
        INCLUDEDIRS "${CMAKE_CURRENT_SOURCE_DIR}"
        MAININCLUDES mdm_status_extension.h
    )
endfunction()
mdmExtensionMain()
